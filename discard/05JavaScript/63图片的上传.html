# 63 图片的上传

# title: 图片的上传

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>图片的上传</title>
</head>
<body>
<label>上传图片的格式验证</label>
<input type="file">
<script>
    var file = document.getElementsByTagName('input')[0];
    file.onchange=function()
    {
        //获取路径
        var path = file.value;
        //console.log(path);
        //截取
        var houzhui =  path.slice(path.lastIndexOf('.'));
        //转化为小写
        var xiaoxie = houzhui.toLowerCase();
        //判断
        if(xiaoxie === '.jpeg' || xiaoxie === '.png' || xiaoxie === '.jpg')
        {
            alert('文件可以上传');
        }
        else{
            alert('文件不可以上传');
        }
    }

</script>
</body>
</html>
