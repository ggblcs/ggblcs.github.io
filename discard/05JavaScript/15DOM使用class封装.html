{
    var aReturn=[];
    var aEle= oParent.getElementsByTagName('*');
    for(var i=0;i<aEle.length;i++)
    {
        if(aEle[i].className==cc)
        {
            aReturn.push(aEle[i]);
        }
    }
    return aReturn;
}
