# 16 继承

  2. 方法的重写
class Animals:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def sayName(self):
        print(f'my name is {self.name}')
    
    def sayAge(self):
        print(f"I am {self.age} years old")
    
class Dog(Animals):
    def __init__(self, name, age):
        # 调用父类的构造方法，初始化在父类_init_方法中定义的所有属性
        # 翻译superclass 超类
        super().__init__(name, age)

    def say(self):
        print('wang wang')

    # 重写父类的方法，Python将不会再考虑这个父类的方法，而只关心在子类中定义的方法
    def sayName(self):
        print(f'I am a dog, my name is {self.name}')

dog = Dog('zs', 18)
dog.say() # wang wang
dog.sayName() # I am a dog, my name is zs
dog.sayAge() # I am 18 years old

class Animals:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def sayName(self):
        print(f'my name is {self.name}')
    
    def sayAge(self):
        print(f"I am {self.age} years old")
    
class Dog(Animals):
    def __init__(self, name, age, color):
        # 调用父类的构造方法，初始化在父类_init_方法中定义的所有属性
        # 翻译superclass 超类
        super().__init__(name, age)
        self.color = color

    def say(self):
        print('wang wang')

    # 重写父类的方法，Python将不会再考虑这个父类的方法，而只关心在子类中定义的方法
    def sayName(self):
        print(f'I am a dog, my name is {self.name}')

    def sayColor(self):
        print(f"my color is {self.color}")

class Person(Animals):
    def __init__(self, name, age):
        super().__init__(name, age)
        self.pet = Dog('wang cai', 2, 'black')
    
    def sayPet(self):
        print(f'I have a pet ,is a dog , Its name is {self.pet.name}')
